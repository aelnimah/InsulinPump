CXX = g++
CXXFLAGS = -std=c++17 -Iinclude -Wall
SRC_DIR = src
OBJ_DIR = obj
BIN = pump_simulator

SOURCES := $(filter-out $(SRC_DIR)/MergedMainWindow.cpp, $(wildcard $(SRC_DIR)/*.cpp))

OBJECTS := $(SOURCES:.cpp=.o)

all: $(BIN)

$(BIN): $(OBJECTS)
	$(CXX) $(CXXFLAGS) -o $@ $^

%.o: %.cpp
	$(CXX) $(CXXFLAGS) -c $< -o $@

clean:
	rm -f $(BIN) $(OBJECTS)
